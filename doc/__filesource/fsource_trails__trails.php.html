  36.  *

  107.   /**
  108.    * Maps a string to a response which is then rendered.
  109.    *
  110.    * @param string The requested URI.
  111.    *
  112.    * @return void 
  113.    */
  114.   function dispatch($uri) {
  115.  
  116.     # E_USER_ERROR|E_USER_WARNING|E_USER_NOTICE|E_RECOVERABLE_ERROR = 5888
  117.     $old_handler = set_error_handler(array($this, 'error_handler'), 5888);
  118.  
  119.     ob_start();
  120.     $level = ob_get_level();
  121.  
  122.     $this->map_uri_to_response($this->clean_request_uri((string) $uri))->output();
  123.  
  124.     while (ob_get_level() >= $level) {
  125.       ob_end_flush();
  126.     }
  127.  
  128.     if (isset($old_handler)) {
  129.       set_error_handler($old_handler);
  130.     }
  131.   }
  132.

  134.   /**
  135.    * Maps an URI to a response by figuring out first what controller to
  136.    * instantiate, then delegating the unconsumed part of the URI to the
  137.    * controller who returns an appropriate response object or throws a
  138.    * Trails_Exception.
  139.    *
  140.    * @param  string  the URI string
  141.    *
  142.    * @return mixed   a response object
  143.    */
  144.   function map_uri_to_response($uri) {
  145.  
  146.     try {
  147.  
  148.       if ('' === $uri) {
  149.         if (!$this->file_exists($this->default_controller . '.php')) {
  150.           throw new Trails_MissingFile(
  151.             "Default controller '{$this->default_controller}' not found'");
  152.         }
  153.         $controller_path = $this->default_controller;
  154.         $unconsumed = $uri;
  155.       }
  156.  
  157.       else {
  158.         list($controller_path, $unconsumed) = $this->parse($uri);
  159.       }
  160.  
  161.       $controller = $this->load_controller($controller_path);
  162.  
  163.       $response = $controller->perform($unconsumed);
  164.  
  165.     } catch (Exception $e) {
  166.  
  167.       $response = isset($controller) ? $controller->rescue($e)
  168.                                      : $this->trails_error($e);
  169.     }
  170.  
  171.     return $response;
  172.   }

  174.   function trails_error($exception) {
  175.     ob_clean();
  176.  
  177.     # show details for local requests
  178.     $detailed = $_SERVER['REMOTE_ADDR'] === '127.0.0.1';
  179.  
  180.     $body = sprintf('<html><head><title>Trails Error</title></head>'.
  181.                     '<body><h1>%s</h1><pre>%s</pre></body></html>',
  182.                     htmlentities($exception->__toString()),
  183.                     $detailed
  184.                       ? htmlentities($exception->getTraceAsString())
  185.                       : '');
  186.  
  187.     if ($exception instanceof <a href="../trails/Trails_Exception.html">Trails_Exception</a>) {
  188.       $response = new <a href="../trails/Trails_Response.html">Trails_Response</a>($body,
  189.                                       $exception->headers,
  190.                                       $exception->getCode(),
  191.                                       $exception->getMessage());
  192.     }
  193.     else {
  194.       $response = new <a href="../trails/Trails_Response.html">Trails_Response</a>($body, array(), 500,
  195.                                       $exception->getMessage());
  196.     }
  197.  
  198.     return $response;
  199.   }

  202.   /**

  209.   function clean_request_uri($uri) {
  210.     if (FALSE !== ($pos = strpos($uri, '?'))) {

  217.   /**
  218.    * <MethodDescription>
  219.    *
  220.    * @param  type       <description>
  221.    * @param  type       <description>
  222.    *
  223.    * @return type       <description>
  224.    */
  225.   function parse($unconsumed, $controller = NULL) {
  226.     list($head, $tail) = $this->split_on_first_slash($unconsumed);
  227.  
  228.     if (!preg_match('/^\w+$/', $head)) {
  229.       throw new <a href="../trails/Trails_RoutingError.html">Trails_RoutingError</a>("No route matches '$head'");
  230.     }
  231.  
  232.     $controller = (isset($controller) ? $controller . '/' : '') . $head;
  233.  
  234.     if ($this->file_exists($controller . '.php')) {
  235.       return array($controller, $tail);
  236.     }
  237.     else if ($this->file_exists($controller)) {
  238.       return $this->parse($tail, $controller);
  239.     }
  240.  
  241.     throw new <a href="../trails/Trails_RoutingError.html">Trails_RoutingError</a>("No route matches '$head'");
  242.   }

  244.   function split_on_first_slash($str) {
  245.     preg_match(":([^/]*)(/+)?(.*):", $str, $matches);
  246.     return array($matches[1], $matches[3]);
  247.   }

  463. /**
  464.  * A Trails_Controller is responsible for matching the unconsumed part of an URI
  465.  * to an action using the left over words as arguments for that action. The
  466.  * action is then mapped to method of the controller instance which is called
  467.  * with the just mentioned arguments. That method can send the #render_action,
  468.  * #render_template, #render_text, #render_nothing or #redirect method.
  469.  * Otherwise the #render_action is called with the current action as argument.
  470.  * If the action method sets instance variables during performing, they will be
  471.  * be used as attributes for the flexi-template opened by #render_action or
  472.  * #render_template. A controller's response's body is populated with the output
  473.  * of the #render_* methods. The action methods can add additional headers or
  474.  * change the status of that response.

  547.     # send action to controller

  569.     return $this->response;
